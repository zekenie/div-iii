exports.index = function(req,res) {
  clientPool.acquire(function(err,mysql) {
    mysql.query("SELECT `user`.*, `match`.* FROM `user`,`match` WHERE `user`.`userID` = `match`.`mUserID` AND `match`.`userID` = ? ORDER BY `match` DESC",[req.session.userID],function(err,matches) {
      if(err) {
        console.log(err);
        clientPool.releaseConnection(mysql);
        return false;
      }
      clientPool.releaseConnection(mysql);
      res.render("matches",{matches:matches});
    });
  });
}
